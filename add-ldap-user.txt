cat > /tmp/add-admin.ldif <<EOF
dn: uid=admin,ou=people,dc=example,dc=org
objectClass: inetOrgPerson
cn: admin
sn: admin
uid: admin
userPassword: admin
EOF

ldapadd -x -D "cn=admin,dc=example,dc=org" -w admin -f /tmp/add-admin.ldif

ldapsearch -x -H ldap://localhost -b "ou=people,dc=example,dc=org" -D "cn=admin,dc=example,dc=org" -w admin "(uid=admin)"
